\titre{ Résolution d'un système linéaire }
\theme{polynômes}
\auteur{}
\organisation{AMSCC}

\contenu{
    \texte{
        On considère le système de trois équations à trois inconnues $x,y,z$ suivant : 
        $$
\left\{\begin{aligned}
m x+m y+m z & =a \\
x+m y+z & =b \\
x+y+m z & =c
\end{aligned}\right.
$$
avec $m,a,b,c \in \R$.
    }
\begin{enumerate}
    \item Supposons que $m=0$. Résoudre le système en fonction des valeurs possibles de $a,b,c$.
    \item Supposons que $m = 1$. \`A quelles conditions sur $a,b,c$ le système admet-il au moins une solution ? Dans ce cas, résoudre le système.
    \item Supposons que $m \neq 0$ et $m \neq 1$. Justifier que le système admet une unique solution quelles que soient les valeurs de $a,b,c$. Déterminer cette solution pour $a=1$, $b=2$, $c=3$ puis pour $a,b,c$ quelconques.
\end{enumerate}
}
