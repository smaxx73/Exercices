\contenu{

Soit la série entière d'une variable réelle définie par :
$$f \colon x \mapsto \sum\limits_{n = 1}^{ + \infty } {\sin \left( {\frac{1}{{\sqrt n }}} \right)x^n } .$$

\begin{enumerate}
	\item \question{ Donner un équivalent de $\sin\left(\frac{1}{\sqrt{n}}\right)$ quand $n \to +\infty$. }
	\item \question{ Déterminer le rayon de convergence $R$ de la série entière $f$. }
	\item \question{ La série est-elle convergente lorsque $x=R$ ? $x=-R$ ? En déduire le domaine de convergence de $f$. }
\end{enumerate}
}
