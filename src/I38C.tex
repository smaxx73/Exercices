\texte{
    Pour chacune des séries suivantes, déterminer si elle converge ou diverge. 
}

\begin{enumerate}
    \item \question{ $\displaystyle \sum_{n\geq 1} \frac{1}{n^2+1}$ ; } 
          \reponse{ On pose $u_n = \frac{1}{n^2+1}$. Alors $u_n \geq 0$ pour tout $n \in \N$ et $u_n \underset{+\infty}\sim \frac{1}{n^2}$ donc par comparaison à une série de Riemann convergente ($\alpha = 2>1$), la série $\sum u_n$ converge. }
    \item \question{ $\displaystyle \sum_{n\geq 1} \frac{|\cos(n)|}{n^2(\sqrt{n}+1)}$ ; }
          \reponse{ On pose $u_n = \frac{|\cos(n)|}{n^2(\sqrt{n}+1)}$. Alors $u_n \geq 0$ pour tout $n \in \N$ et $u_n \leq \frac{1}{n^2(\sqrt{n}+1)} \underset{+\infty}\sim \frac{1}{n^{5/2}}$ pour tout $n \in \N$ donc par comparaison d'une série positive à une série de Riemann convergente, la série $\sum u_n$ converge. }
\end{enumerate}}
