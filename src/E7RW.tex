\contenu{

\question{ Résoudre le système d'équations : $$\left\{\begin{array}{l}3 x_1+4 x_2+x_3+2 x_4=3 \\ 6 x_1+8 x_2+2 x_3+5 x_4=7 \\ 9 x_1+12 x_2+3 x_3+10 x_4=13+a\end{array} \quad a \in \mathbb{R}\right.$$ }


\reponse{ On a :
$$
\begin{aligned}
& \left\{\begin{aligned}
3 x_1+4 x_2+x_3+2 x_4 & =3 \\
6 x_1+8 x_2+2 x_3+5 x_4 & =7 \\
9 x_1+12 x_2+3 x_3+10 x_4 & =13+a
\end{aligned} \Leftrightarrow \begin{array}{rl}
\ell_1 \\
\ell_2-2 \ell_1 \\
\ell_3-3 \ell_1
\end{array}\right. \left\{ \begin{aligned}
3 x_1+4 x_2+x_3+2 x_4 & =3 \\
x_4 & =1 \\
4 x_4 & =4+a
\end{aligned} \right. \\
& \Leftrightarrow\left\{\begin{aligned}
3 x_1+4 x_2+x_3+2 x_4 & =3 \\
x_4 & =1 \\
0 & =a
\end{aligned}\right. \\
&
\end{aligned}
$$
- Si $a \neq 0$, le système n'a pas de solution ;
- Si $a=0$, le système est équivalent à :
$$
\left\{\begin{aligned}
3 x_1+4 x_2+x_3+2 x_4 & =3 \\
x_4 & =1
\end{aligned}\right.
$$
et l'ensemble des solutions est $$\left\{\left(x_1, x_2, 1-3 x_1-4 x_2, 1\right) \, \mid \, x_1 \in \mathbb{R}, x_2 \in \mathbb{R}\right\}$$ }}
