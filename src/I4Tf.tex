\contenu{

\question{ 	Déterminer si les ensembles suivants sont ouverts ou fermés dans $\R$.
	
	$$[0;1[ \, , \, [0;+\infty[ \, , \, ]0;1[\cup \{2\} \, , \, \Z \, , \, \left\{\frac{1}{n} \in \R \, \mid n \in \N^* \right\} $$ }
\reponse{
\begin{itemize}
	\item $[0;1[$ est ni ouvert, ni fermé ;
	\item $[0;+\infty[$ est fermé ;
	\item $]0;1[\cup \{2\}$ est ni ouvert, ni fermé ;
	\item $\Z$ est fermé ;
	\item $\left\{\frac{1}{n} \in \R \, \mid n \in \N^* \right\}$ est ni ouvert, ni fermé.
\end{itemize}
}}
