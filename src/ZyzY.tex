\contenu{
    \texte{
        Soient les polynômes $P,Q \in \R[X]$ définis par $P(X) = -6X^3 + 4X^2 + X^4 + X^3 + 3X + 9$ et $Q(X) = X^2 + X + 1$. 
    }
\begin{enumerate}
    \item \question{ Donner le degré des polynômes $P$ et $Q$. }
    \item \question{ Montrer que $3$ est une racine double de $P$. }
    \item \question{ Déterminer l'ensemble des racines réelles de $Q$ puis l'ensemble des racines complexes de $Q$. En déduire une factorisation de $Q$ en polynômes irréductibles dans $\mathbb{C}[X]$. }
    \item \question{ Effectuer la division euclidienne de $P$ par $Q$. }
    \item \question{ En déduire une factorisation de $P$ en polynômes irréductibles dans $\mathbb{C}[X]$. }
\end{enumerate}
}
