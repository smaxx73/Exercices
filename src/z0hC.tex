\contenu{

\texte{ On considère les matrices: $A=\left(\begin{array}{lll}1 & 0 & 0 \\ 0 & 1 & 1 \\ 3 & 1 & 1\end{array}\right), B=\left(\begin{array}{lll}1 & 1 & 1 \\ 0 & 1 & 0 \\ 1 & 0 & 0\end{array}\right)$ et $C=\left(\begin{array}{ccc}1 & 1 & 1 \\ 1 & 2 & 1 \\ 0 & -1 & -1\end{array}\right)$. }

\begin{enumerate}
	\item \question{  Calculer les produits $AB$ et $AC$. Que constate-t-on ? }
	\item \question{ La matrice $A$ est-elle inversible ? }
\end{enumerate}

\reponse{ On a :
	$$
	\begin{aligned}
		& A \cdot B=\left(\begin{array}{lll}
			1 & 0 & 0 \\
			0 & 1 & 1 \\
			3 & 1 & 1
		\end{array}\right) \cdot\left(\begin{array}{ccc}
			1 & 1 & 1 \\
			0 & 1 & 0 \\
			1 & 0 & 0
		\end{array}\right)=\left(\begin{array}{lll}
			1 & 1 & 1 \\
			1 & 1 & 0 \\
			4 & 4 & 3
		\end{array}\right) \\
		& A \cdot C=\left(\begin{array}{lll}
			1 & 0 & 0 \\
			0 & 1 & 1 \\
			3 & 1 & 1
		\end{array}\right) \cdot\left(\begin{array}{ccc}
			1 & 1 & 1 \\
			1 & 2 & 1 \\
			0 & -1 & -1
		\end{array}\right)=\left(\begin{array}{lll}
			1 & 1 & 1 \\
			1 & 1 & 0 \\
			4 & 4 & 3
		\end{array}\right)
	\end{aligned}
	$$
	On remarque que :
	$$
	A \cdot B=A \cdot C
	$$
	Montrons que la matrice $A$ n'est pas inversible.
	(Démonstration par l'absurde.)
	Supposons que $A$ soit inversible, notons $A^{-1}$ son inverse.
	On aurait :
	$$
	A^{-1} \cdot A \cdot B=A^{-1} \cdot A \cdot C \Leftrightarrow B=C
	$$
	Or $B \neq C$. Donc $A$ n'est pas inversible. }}
