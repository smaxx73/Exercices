\chapitre{Géométrie affine euclidienne}
\sousChapitre{Géométrie affine euclidienne du plan}

\contenu{
\texte{
Soit $E$ un plan euclidien orienté muni d'un repère orthonormé direct $(0,\vec{I},\vec{J})$.
Soit $\vec{u}$ un vecteur de norme $1$ et de coordonnées $\begin{pmatrix}X\\Y\end{pmatrix}$ dans le repère 
$(0,\vec{I},\vec{J})$.
}
\begin{enumerate}
    \item \question{Calculer le produit scalaire $\vec{I}\cdot\vec{u}$.}
    \item \question{Calculer, à l'aide de la définition de la fonction cosinus, la quantité
$||\vec{I}||\ ||\vec{u}||\cos\widehat{(\vec{I},\vec{u})}$.}
    \item \question{Reprendre les calculs précédents, sans l'hypothèse que $\vec{u}$ est de norme $1$.}
\end{enumerate}
}
