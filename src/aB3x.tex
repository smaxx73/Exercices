\contenu{
\texte{
  Représenter graphiquement les sous-ensembles suivants de $\mathbb{R}^2$. Lesquels sont des ouverts ? Aucune justification n'est demandée.
}
\begin{enumerate}
  \item \question{$A = \mathbb{R}^*_+ \times \mathbb{R}^*_+$}
    \reponse{$A$ est ouvert.}

  \item \question{$B = [0,1] \times [0,1]$}
    \reponse{$B$ est fermé.}

  \item \question{$C = ]0,1[ \times ]0,1[$}
    \reponse{$C$ est ouvert.}

  \item \question{$D = \mathbb{R}^* \times \mathbb{R}$}
    \reponse{$D$ est ouvert.}

  \item \question{$E = \mathbb{R} \times \mathbb{R}_+$}
    \reponse{$E$ est fermé.}

  \item \question{$F = ]1,+\infty[ \times \mathbb{R}^*_+$}
    \reponse{$F$ est ouvert.}

  \item \question{$G = \mathbb{R}^2 \setminus \{(0,0)\}$}
    \reponse{$G$ est ouvert.}

  \item \question{$H = (\mathbb{R} \setminus \{-1,1\}) \times \mathbb{R}^*_+$}
    \reponse{$H$ est ouvert.}

  \item \question{$I = \{(x,y) \in \mathbb{R}^2 \mid x + y \geq 1\}$}
    \reponse{$I$ est fermé.}
\end{enumerate}
}
