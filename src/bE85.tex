\question{ Calculer les racines carrées de $1, i, 3+4 i, 8-6 i, 7+24 i$. }

\reponse{ \begin{enumerate} \item Les deux racines carrées de $z=1=e^{i .0}$ sont $z_k=e^{i \cdot \frac{0}{2}+i k \pi} \quad k \in\{0 ; 1\}$, c'est-à-dire : $z_0=e^{i .0}=1$ et $z_1=e^{i . \pi}=-1$;
	\item Les deux racines carrées de $z=i=e^{i \cdot \frac{\pi}{2}}$ sont $z_k=e^{i \cdot \frac{\pi}{4}+i k \pi} \quad k \in\{0 ; 1\}$, c'est-à-dire $: z_0=e^{i \cdot \frac{\pi}{4}}$ et $z_1=e^{i \frac{\pi}{4}+i \pi}=e^{i \cdot \frac{5 \pi}{4}}$
	\item Les deux racines carrées de $z=3+4 i=a+i b$ sont $z_1=x+i y$ et $z_2=-x$-iy telles que : $(x+i y)^2=x^2-y^2+2 i x y=3+4 i \quad$ et $\quad|x+i y|^2=|3+4 i| \Leftrightarrow x^2+y^2=\sqrt{3^2+4^2}=5$. Résolvons :
	$$
	\left\{\begin{array} { r l } 
		{ x ^ { 2 } - y ^ { 2 } } & { = 3 } \\
		{ 2 x y } & { = 4 > 0 } \\
		{ x ^ { 2 } + y ^ { 2 } } & { = 5 }
	\end{array} \Leftrightarrow \left\{\begin{array}{l}
		x^2=\frac{3+5}{2}=4 \\
		y^2=\frac{5-3}{2}=1 \\
		x y>0
	\end{array}\right.\right.
	$$
	Les deux carrées de $z=3+4 i$ sont donc $z_1=2+i$ et $z_1=-2-i$.
	\item Les deux racines carrées de $z=8-6 i=a+i b$ sont $z_1=x+i y$ et $z_2=-x-i y$ telles que : $(x+i y)^2=x^2-y^2+2 i x y=8-6 i \quad$ et $\quad|x+i y|^2=|8-6 i| \Leftrightarrow x^2+y^2=\sqrt{8^2+6^2}=10$. Résolvons :
	$$
	\left\{\begin{array} { r l } 
		{ x ^ { 2 } - y ^ { 2 } } & { = 8 } \\
		{ 2 x y } & { = - 6 < 0 } \\
		{ x ^ { 2 } + y ^ { 2 } } & { = 1 0 }
	\end{array} \Leftrightarrow \left\{\begin{array}{l}
		x^2=\frac{8+10}{2}=9 \\
		y^2=\frac{10-8}{2}=1 \\
		x y<0
	\end{array}\right.\right.
	$$
	Les deux carrées de $z=8-6 i$ sont donc $z_1=3-i$ et $z_1=-3+i$.
	\item $7+24 i$ Les deux racines carrées de $z=7+24 i=a+i b$ sont $z_1=x+i y$ et $z_2=-x-i y$ telles que : $(x+i y)^2=x^2-y^2+2 i x y=7+24 i$ et :
	$$
	|x+i y|^2=|7+24 i| \Leftrightarrow x^2+y^2=\sqrt{7^2+24^2}=25 .
	$$
	Résolvons :
	$$
	\left\{\begin{array} { r l } 
		{ x ^ { 2 } - y ^ { 2 } } & { = 7 } \\
		{ 2 x y } & { = 2 4 < 0 } \\
		{ x ^ { 2 } + y ^ { 2 } } & { = 2 5 }
	\end{array} \Leftrightarrow \left\{\begin{array}{rl}
		x^2=\frac{7+25}{2}=16 \\
		y^2=\frac{25-7}{2}=9 \\
		x y>0
	\end{array}\right.\right.
	$$
	Les deux carrées de $z=7+24 i$ sont donc $z_1=4+3 i$ et $z_1=-4-3 i$.
	\end{enumerate}
 }}
