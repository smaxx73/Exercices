\texte{ On considère les matrices suivantes:
$$
A=\left(\begin{array}{lll}
	1 & 2 & 3
\end{array}\right) \quad B=\left(\begin{array}{c}
	1 \\
	-2
\end{array}\right) \quad C=\left(\begin{array}{cc}
	2 & 1 \\
	-3 & 0 \\
	1 & 2
\end{array}\right)$$ 
$$D=\left(\begin{array}{cc}
	-2 & 5 \\
	5 & 0
\end{array}\right) \quad E=\left(\begin{array}{ccc}
	-1 & 1 & 3 \\
	-1 & -4 & 0 \\
	0 & 2 & 5
\end{array}\right)
$$ }

\begin{enumerate}
	\item \question{ Quelles sont les matrices carrées et les matrices symétriques ? }
	\item \question{ Quels sont les produits matriciels (de 2 matrices) possibles ? }
\end{enumerate}
 
 \reponse{ $A$ est une matrice $1 \times 3, B$ est une matrice $2 \times 1, C$ est une matrice $3 \times 2, D$ est une matrice $2 \times 2$ et $E$ est une matrice $3 \times 3$.
 	Seules $D$ et $E$ sont des matrices carrées.
 	De ces deux matrices carrées, la seule qui est symétrique est la matrice $D$.
 	
 	Les produits possibles sont :
 \begin{enumerate}
 	\item $A\times C$ et la matrice produit est de type $1 \times 2$ ;
 	\item $A\times E$ et la matrice produit est de type $1 \times 3$ ;
 	\item $B\times A$ et la matrice produit est de type $2 \times 3$ ;
 	\item $C \times B$ et la matrice produit est de type $3 \times 1$ ;
 	\item $C\times D$ et la matrice produit est de type $3 \times 2$ ;
 	\item $D \times B$ et la matrice produit est de type $2 \times 1$ ;
 	\item $D\times D = D^2$ et la matrice produit est de type $2 \times 2$ ;
 	\item $E \times C$ et la matrice produit est de type $3 \times 2$ ;
 	\item $E \times E$ et la matrice produit est de type $3 \times 3$ ;
 \end{enumerate}
  }}
