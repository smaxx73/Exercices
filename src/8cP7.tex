\texte{ 
On considﾃｨre la fonction $f:\mathbb{R}^2\to\mathbb{R}$ dﾃｩfinie par
\[

\begin{cases}
\dfrac{x^2y}{x^2+y^2} & \text{si } (x,y)\neq(0,0),\\[1ex]

\end{cases}
\]

\begin{enumerate}

\item   \question{ﾃ液udier la continuitﾃｩ de $f$ sur $\mathbb{R}^2$.}
\reponse{
Sur $\mathbb{R}^2\setminus\{(0,0)\}$, la fonction est quotient de fonctions polynomiales avec dﾃｩnominateur non nul, donc elle est continue.

Il reste ﾃ ﾃｩtudier la continuitﾃｩ en $(0,0)$. Pour $(x,y)\neq(0,0)$,
\[
|f(x,y)|=\left|\frac{x^2y}{x^2+y^2}\right|
\le |y|\frac{x^2}{x^2+y^2}
\le |y|.
\]
Comme $(x,y)\to(0,0)$ implique $|y|\to 0$, on obtient
\[
f(x,y)\to 0=f(0,0).
\]
Donc $f$ est continue sur $\mathbb{R}^2$.
}

\item   \question{Calculer les dﾃｩrivﾃｩes partielles de $f$ en $(0,0)$, si elles existent.}
\reponse{
Par dﾃｩfinition,
\[
\frac{\partial f}{\partial x}(0,0)
=
\lim_{h\to 0}\frac{f(h,0)-f(0,0)}{h}.
\]
Or
\[

\]

\[
\frac{\partial f}{\partial x}(0,0)=0.
\]

De mﾃｪme,
\[
\frac{\partial f}{\partial y}(0,0)
=
\lim_{h\to 0}\frac{f(0,h)-f(0,0)}{h}.
\]
Or
\[
f(0,h)=0,
\]
donc
\[
\frac{\partial f}{\partial y}(0,0)=0.
\]

Les deux dﾃｩrivﾃｩes partielles existent en $(0,0)$ et valent $0$.
}

\item   \question{Calculer les dﾃｩrivﾃｩes partielles de $f$ sur $\mathbb{R}^2\setminus\{(0,0)\}$.}
\reponse{
Pour $(x,y)\neq(0,0)$, on dﾃｩrive par la formule du quotient.

On obtient
\[
\frac{\partial f}{\partial x}(x,y)
=
\frac{(2xy)(x^2+y^2)-x^2y(2x)}{(x^2+y^2)^2}
=
\frac{2xy^3}{(x^2+y^2)^2},
\]
et
\[
\frac{\partial f}{\partial y}(x,y)
=
\frac{x^2(x^2+y^2)-x^2y(2y)}{(x^2+y^2)^2}
=
\frac{x^2(x^2-y^2)}{(x^2+y^2)^2}.
\]
}

\item   \question{La fonction $f$ est-elle de classe $C^1$ sur $\mathbb{R}^2$ ?}
\reponse{
Sur $\mathbb{R}^2\setminus\{(0,0)\}$, $f$ est de classe $C^\infty$.

Il faut ﾃｩtudier la continuitﾃｩ des dﾃｩrivﾃｩes partielles en $(0,0)$.

Pour
\[
\frac{\partial f}{\partial x}(x,y)=\frac{2xy^3}{(x^2+y^2)^2},
\]
on a
\[
\left|\frac{\partial f}{\partial x}(x,y)\right|
\le \frac{2|x||y|^3}{(x^2+y^2)^2}.
\]
En coordonnﾃｩes polaires $x=r\cos\theta$, $y=r\sin\theta$, cela devient
\[

\]
ce qui ne tend pas vers $0$ quand $r\to 0$ en gﾃｩnﾃｩral. Donc $\frac{\partial f}{\partial x}$ n窶册st pas continue en $(0,0)$.

Ainsi $f$ n窶册st pas de classe $C^1$ sur $\mathbb{R}^2$.
}

\item   \question{La fonction $f$ est-elle diffﾃｩrentiable en $(0,0)$ ?}
\reponse{
Si $f$ ﾃｩtait diffﾃｩrentiable en $(0,0)$, sa diffﾃｩrentielle serait nulle puisque les dﾃｩrivﾃｩes partielles y valent $0$. Il faudrait alors
\[

\]

Or en prenant $y=x$ avec $x\neq 0$,
\[
f(x,x)=\frac{x^3}{2x^2}=\frac{x}{2}.
\]
Comme
\[
\|(x,x)\|=\sqrt{2}|x|,
\]
on a
\[
\frac{|f(x,x)|}{\|(x,x)\|}
=
\frac{|x|/2}{\sqrt{2}|x|}

\]
ce qui ne tend pas vers $0$.

Donc $f$ n窶册st pas diffﾃｩrentiable en $(0,0)$.
}

\end{enumerate}
