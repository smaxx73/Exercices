\contenu{

	A chaque fonction définie ci-dessous (1-6), associer sa surface représentative (A-F).

\begin{multicols}{3}
	\begin{enumerate}
		\item $f(x,y) = |x|+|y|$
		\item $f(x,y) = |xy|$
		\item $f(x,y) = \frac{1}{1+x^2+y^2}$
		\item $f(x,y) = (x^2-y^2)^2$
		\item $f(x,y) = (x-y)^2$
		\item $f(x,y) = \sin(|x|+|y|)$
	\end{enumerate}
\end{multicols}
