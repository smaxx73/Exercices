\begin{enumerate}
    \item \question{Déterminer si possible un plan totalement isotrope pour chacune des formes bilinéaires symétriques données par les formes quadratiques suivantes sur $\mathbb{C}^4$.
$$P(x,y,z,t)=xy+zt \textrm{ et } Q(x,y,z)=x^2+y^2+z^2+t^2.$$}
    \item \question{L'entier $-3$ est-il un carré modulo 7 ?}
    \item \question{Déterminer si possible un vecteur non-nul isotrope pour chacune des formes bilinéaires symétriques données par les formes quadratiques suivantes sur $\mathbb{F}_7^4$.
$$R(x,y,z,t)=xy+zt\textrm{ et } S(x,y,z)=x^2+y^2+z^2+t^2.$$}
\end{enumerate}
