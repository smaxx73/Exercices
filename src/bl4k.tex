\question{ Calculer les déterminants ci-dessous en indiquant la méthode de calcul :

$$
\Delta_1=\begin{vmatrix}
	2 & 3 & -1 & 3 \\
	1 & 1 & 2 & 1 \\
	-1 & 2 & 1 & 1 \\
	2 & -1 & 1 & -1
\end{vmatrix} \quad \Delta_2=\begin{vmatrix}
	x & x & x & x & x \\
	-x & 0 & x & x & x \\
	-x & -x & 0 & x & x \\
	-x & -x & -x & 0 & x \\
	-x & -x & -x & -x & 0
\end{vmatrix}
$$ }

\reponse{ 
$$
\begin{aligned}
	\Delta_1 & =\left|\begin{array}{cccc}
		2 & 3 & -1 & 3 \\
		1 & 1 & 2 & 1 \\
		-1 & 2 & 1 & 1 \\
		2 & -1 & 1 & -1
	\end{array}\right|=-\left|\begin{array}{cccc}
		1 & 1 & 2 & 1 \\
		2 & 3 & -1 & 3 \\
		-1 & 2 & 1 & 1 \\
		2 & -1 & 1 & -1
	\end{array}\right| \quad\left(\begin{array}{l}
		l_1^{\prime}=l_2 \\
		l^{\prime}{ }_2=l_1 \\
		l_3^{\prime}=l_3 \\
		l_4^{\prime}=l_4
	\end{array}\right) \\
	& \left.=-\left|\begin{array}{cccc}
		1 & 1 & 2 & 1 \\
		0 & 1 & -5 & 1 \\
		0 & 3 & 3 & 2 \\
		0 & -3 & -3 & -3
	\end{array}\right| \quad  \begin{array}{ll}
		l^{\prime \prime}{ }_1=l_1^{\prime} \\
		l_1^{\prime \prime}=l^{\prime}{ }_2-2 l_1^{\prime} \\
		l^{\prime \prime}{ }_3=l_3^{\prime}+l_1^{\prime} \\
		l^{\prime \prime}{ }_4=l^{\prime}{ }_4-2 l^{\prime}{ }_1
	\end{array}\right) \\
	& =3 \cdot\left|\begin{array}{cccc}
		1 & 1 & 2 & 1 \\
		0 & 1 & -5 & 1 \\
		0 & 3 & 3 & 2 \\
		0 & 1 & 1 & 1
	\end{array}\right|=3 \cdot\left|\begin{array}{ccc}
		1 & -5 & 1 \\
		3 & 3 & 2 \\
		1 & 1 & 1
	\end{array}\right|=3 .(-3-2+15+3+3-10)=3.6 \\
	& =18
\end{aligned}
$$
$$
\begin{aligned}
	\Delta_2 & =\left|\begin{array}{ccccc}
		x & x & x & x & x \\
		-x & 0 & x & x & x \\
		-x & -x & 0 & x & x \\
		-x & -x & -x & 0 & x \\
		-x & -x & -x & -x & 0
	\end{array}\right|=\left|\begin{array}{ccccc}
		C_1 & C_2-C_1 & C_3-C_1 & C_4-C_1 & C_5-C_1 \\
		-x & 0 & 0 & 0 & 0 \\
		-x & 0 & 2 x & 2 x & 2 x \\
		-x & 0 & 0 & 2 x & 2 x \\
		-x & 0 & 0 & 0 & x
	\end{array}\right| \\
	& =x \cdot\left|\begin{array}{cccc}
		x & 2 x & 2 x & 2 x \\
		0 & x & 2 x & 2 x \\
		0 & 0 & x & 2 x \\
		0 & 0 & 0 & x
	\end{array}\right| \\
	& =x^5
\end{aligned}
$$
 }}
