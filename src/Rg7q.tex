\begin{enumerate}
\item   \question{Calculer $I_1=\displaystyle\int_0^{\pi/3}\bigl(1-\cos(3x)\bigr)\,dx$.}
\indication{}
\reponse{$\displaystyle I_1=\left[x-\frac13\sin(3x)\right]_0^{\pi/3}=\frac{\pi}{3}-\frac13\sin(\pi)=\frac{\pi}{3}$.}

\item   \question{Calculer $I_2=\displaystyle\int_0^{\sqrt{\pi}} 2x\sin(x^2)\,dx$.}
\indication{}
\reponse{Posons $u=x^2$, $du=2x\,dx$. Alors $I_2=\displaystyle\int_0^{\pi}\sin u\,du=\bigl[-\cos u\bigr]_0^{\pi}=2$.}

\item   \question{Calculer $I_3=\displaystyle\int_1^{2}\frac{1}{x\sqrt{\ln x}}\,dx$.}
\indication{}
\reponse{Posons $u=\ln x$, $du=\frac{dx}{x}$. Alors $I_3=\displaystyle\int_{0}^{\ln 2} u^{-1/2}du=\left[2\sqrt{u}\right]_0^{\ln 2}=2\sqrt{\ln 2}$.}

\end{enumerate}
