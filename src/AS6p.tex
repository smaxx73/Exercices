\contenu{

\texte{ Pour tout $n \in \N$, on pose $u_n = \frac{5}{4^n}$.  }

\question{ Montrer que la série $\displaystyle \sum_{n\geq 0} u_n$ converge et calculer sa somme. }

\reponse{ On a $u_n = 5\left(\frac{1}{4}\right)^n$ donc il s'agit d'une série géométrique de raison $q =  \frac{1}{4}$. \\
    Comme $|q| < 1$, la série converge et sa somme est $\frac{u_0}{1-q} = \frac{5}{1-\frac{1}{4}} = \frac{20}{3}$. 
}
}
