\chapitre{Groupe orthogonal et quaternions}
\sousChapitre{Groupe orthogonal et quaternions}

\contenu{
\texte{
On considère la forme symplectique sur $\Rr^{2n}$ donnée par 
$$f\left(\left({x\atop y}\right),\left({x'\atop y'}\right)\right)
=x^t\ y'-y^t\ x'$$
o{ù} $x,y,x',y'$ sont dans $\Rr^n$.
}
\begin{enumerate}
    \item \question{En décomposant par bloc $n\times n$ une matrique quelconque $g$ de
 $M_{2n}(\Rr)$ sous la forme $\left({AB\atop CD}\right)$,
 caractériser les matrices symplectiques en termes de systèmes
 d'équation pour $A,B,C,D$.}
    \item \question{Déterminer toutes les matrices symplectiques vérifiant de plus
 $B=C=0$.}
    \item \question{Déterminer toutes les matrices symplectiques vérifiant
 $A=D=I_n$ et $C=0$.}
    \item \question{Déterminer toutes les matrices symplectiques vérifiant $C=0$.}
    \item \question{Déterminer toutes les matrices $Q$ de $M_{n}(\Rr)$ telles que
 l'espace $W_Q:=\left\{ \left({Qy\atop y}\right)/ y\in
 \Rr^n\right\}$ 
soit totalement isotrope. Quel est le lien avec les questions
 précédentes ?}
\end{enumerate}
}
