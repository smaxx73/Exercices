\contenu{

\question{ 	Déterminer si les ensembles suivants sont ouverts ou fermés dans $\R^2$. }
\begin{itemize}
	\item \question{ $A=\{(x,y)\in \R^2 \mid 0<\vert x-1\vert <1 \}$ ; }
	\reponse{ouvert}
	\item \question{ $B=\{(x,y)\in \R^2 \mid 0\leq x\leq y\}$ ; }
	\reponse{fermé}
	\item \question{ $C=\{(x,y)\in \R^2 \mid \vert x\vert <1,\; \vert y\vert \leq
	1 \}$ }
	\reponse{ni ouvert, ni fermé}
\end{itemize}}
