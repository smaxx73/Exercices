\contenu{
\texte{ 		Soit $u \colon t \mapsto u(t)$ une fonction de classe $\mathcal{C}^1$ sur $\R$ et $h \colon (x,y) \mapsto h(x,y)$ une fonction de classe $\mathcal{C}^1$ sur $\R^2$. Pour tout $t \in \R$, on pose :
$$F(t) = h(u(t),u(t)^2)$$ }
\question{ Vérifier que $F$ est dérivable sur $\R$ et exprimer sa dérivée en fonction de $u(t)$ et $u'(t)$. }

}
